# 2 - 链式调用和延迟执行

```js
arrange('William')
  .wait(5)
  .do('commmit')
  .waitFirst(3)
  .execute();

// 等待3秒
// > William is not defined
// 等待5秒
// > Start to commit
```


```js
function arrange(name) {
            const tasks = [];
            tasks.push(() => {
                console.log(`> ${name} is not notified`);
            })

            function wait(seconds){
                tasks.push(()=>new Promise(resolve => {
                    setTimeout(resolve, seconds * 1000);
                }))
                return this;
            }

            function doAction(action){
                tasks.push(() => {
                    console.log(`> Start to ${action}`);
                })
                return this;
            }

            function waitFirst(seconds){
                tasks.unshift(()=>new Promise(resolve => {
                    setTimeout(resolve, seconds * 1000);
                }))
                return this;
            }

            function execute(){
                (async ()=>{
                    for(const task of tasks){
                        await task();
                    }
                })();
                return this;
            }

            return {
                wait,
                do: doAction,
                waitFirst,
                execute,
            }
        }

        arrange('William')
            .wait(5)
            .do('commmit')
            .waitFirst(3)
            .execute();
```
